pyenv 工程实践指南

安装

brew install pyenv

或者

Python 3.7 依赖 OpenSSL 1.0.2e 或更高版本，手动安装OpenSSL或者直接安装3.6最新版本

 1git clone https://github.com/pyenv/pyenv.git ~/.pyenv
 2echo 'export PYENV_ROOT="$HOME/.pyenv"' >> ~/.bash_profile
 3echo 'export PATH="$PYENV_ROOT/bin:$PATH"' >> ~/.bash_profile
 4echo -e 'if command -v pyenv 1>/dev/null 2>&1; then\n  eval "$(pyenv init -)"\nfi' >> ~/.bash_profile
 5
 6git clone https://github.com/yyuu/pyenv-virtualenv.git ~/.pyenv/plugins/pyenv-virtualenv
 7echo 'eval "$(pyenv virtualenv-init -)"' >> ~/.bash_profile
 8
 9sudo yum install openssl-devel zlib-devel python-devel -y
10
11pyenv install -v 3.6.10
12# 若下载失败则手动下载上传到cache目录再安装
13mkdir ~/.pyenv/cache
14cd ~/.pyenv/cache
15wget https://www.python.org/ftp/python/3.6.10/Python-3.6.10.tgz -P ~/.pyenv/cache
16mv ~/.pyenv/cache/Python-3.6.10.tgz ~/.pyenv/cache/Python-3.6.10.tar.gz
17cp ~/.pyenv/cache/Python-3.6.10.tar.gz /tmp
18
19pyenv install -v 3.6.10
20# 或
21CFLAGS=-I/usr/include/openssl LDFLAGS=-L/usr/lib64 pyenv install -v 3.6.10
22
23# CentOS7 下载
24# https://www.python.org/ftp/python/3.7.6/Python-3.7.6.tar.xz
25# https://www.python.org/ftp/python/3.6.10/Python-3.6.10.tar.xz

安装cx_Oracle

需要保证 Python 位数与 Oracle intstantclient 位数一致

 1sudo rpm -ivh oracle-instantclient12.1-basic-12.1.0.2.0-1.x86_64.rpm
 2# 或
 3sudo yum install oracle-instantclient12.1-basic-12.1.0.2.0-1.x86_64.rpm
 4
 5sudo cp /usr/lib/oracle/12.1/client64/lib/libclntsh.so.12.1 /usr/lib/oracle/12.1/client64/lib/libclntsh.so
 6sudo sh -c "echo /usr/lib/oracle/18.5/client64/lib > /etc/ld.so.conf.d/oracle-instantclient.conf"
 7sudo ldconfig
 8sudo export LD_LIBRARY_PATH=/usr/lib/oracle/12.1/client64/lib:$LD_LIBRARY_PATH
 9sudo mkdir -p /usr/lib/oracle/12.1/client64/lib/network/admin
10sudo export TNS_ADMIN=/usr/lib/oracle/12.1/client64/lib/network/admin
11
12pyenv shell 3.6.10
13# Or
14pyenv virtualenv 3.6.10 venv-db-py3.6
15pyenv shell venv-db-py3.6
16
17pip install -i https://pypi.doubanio.com/simple cx_oracle
